  def util_module_or_class(exp, is_class=false)
    s = "#{exp.shift}"

    if is_class then
      superk = process(exp.shift)
      s << " < #{superk}" if superk
    end

    s << "\n"

    body = []
    begin
      code = process(exp.shift).chomp
      body << code unless code.nil? or code.empty?
    end until exp.empty?
    unless body.empty? then
      body = indent(body.join("\n\n")) + "\n"
    else
      body = ""
    end
    s + body + "end"
  end
