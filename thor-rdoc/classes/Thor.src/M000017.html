  def help
    list = self.class.help_list
    puts "Options"
    puts "-------"
    list.usages.each do |meth, use|
      format = "%-" + (list.max.usage + list.max.opt + 4).to_s + "s"
      print format % ("#{usage(meth)}")
      puts  list.descriptions.assoc(meth)[1]
    end
  end
