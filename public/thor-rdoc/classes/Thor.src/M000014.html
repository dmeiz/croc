  def self.start
    meth = ARGV.shift
    params = []
    while !ARGV.empty?
      break if ARGV.first =~ /^\-/
      params << ARGV.shift
    end
    if defined?(@map) && @map[meth]
      meth = @map[meth].to_s
    end
    
    args = ARGV.dup
    
    if @opts.assoc(meth)
      opts = @opts.assoc(meth).last.map {|opt, val| [opt, val == true ? Getopt::BOOLEAN : Getopt.const_get(val)].flatten}
      options = Getopt::Long.getopts(*opts)
      params << options
    end
    
    ARGV.replace args
    
    new(meth, params).instance_variable_get("@results")
  end
