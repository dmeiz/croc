  def method_missing(meth, delete=false)
    matches = find_all { | sexp | Sexp === sexp and sexp.first == meth }

    case matches.size
    when 0 then
      nil
    when 1 then
      match = matches.first
      delete match if delete
      match
    else
      raise NoMethodError, "multiple nodes for #{meth} were found in #{inspect}"
    end
  end
