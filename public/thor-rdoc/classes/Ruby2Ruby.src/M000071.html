  def process_fcall(exp)
    recv = exp.shift unless Symbol === exp.first # HACK conditional - some not getting rewritten?
    name = exp.shift.to_s
    args = exp.shift
    code = []
    unless args.nil? then
      args[0] = :arglist if args.first == :array
      code << process(args)
    end
    return code.empty? ? name : "#{name}(#{code.join(', ')})"
  end
