  def util_dthing(exp, regx = false)
    s = []
    suck = true
    if suck then
      x = exp.shift.gsub(/"/, '\"').gsub(/\n/, '\n')
    else
      x = exp.shift.dump[1..-2]
    end
    x.gsub!(/\//, '\/') if regx

    s << x
    until exp.empty?
      pt = exp.shift
      case pt
      when Sexp then
        case pt.first
        when :str then
          if suck then
            x = pt.last.gsub(/"/, '\"').gsub(/\n/, '\n')
          else
            x = pt.last.dump[1..-2]
          end
          x.gsub!(/\//, '\/') if regx
          s << x
        else
          s << '#{' << process(pt) << '}' # do not use interpolation here
        end
      else
        # do nothing - yet
      end
    end

    s.join
  end
