  def process_hash(exp)
    result = []
    until exp.empty?
      result << "#{process(exp.shift)} => #{process(exp.shift)}"
    end

    case self.context[1]
    when :arglist, :argscat then
      return "#{result.join(', ')}" # HACK - this will break w/ 2 hashes as args
    else
      return "{ #{result.join(', ')} }"
    end
  end
