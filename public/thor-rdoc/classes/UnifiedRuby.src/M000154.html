  def rewrite_call(exp)
    args = exp.last
    case args
    when nil
      exp.pop
    when Array
      case args.first
      when :array, :arglist then
        args[0] = :arglist
      when :argscat, :splat then
        # do nothing
      else
        raise "unknown type in call #{args.first.inspect}"
      end
      return exp
    end

    exp << s(:arglist)

    exp
  end
